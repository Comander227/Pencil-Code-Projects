## Promting and replacement function
Prompting = (madLib) ->
  for i in [1...madLib.length] by 2
    await read madLib[i], defer answer
    madLib[i] = answer
  typeline()
  typeline()

## Presenting user generated Mad Lib  
  write 'Here is your Mad Lib'
  typeline()
  
  write madLib.join ''
  

## Directions

## User Interface
